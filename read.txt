CREATE TABLE leave_type (
    id INT AUTO_INCREMENT PRIMARY KEY,
    company_id INT NOT NULL,
    code VARCHAR(50) NOT NULL,
    name VARCHAR(100) NOT NULL,
    is_active TINYINT(1) NOT NULL DEFAULT 1,

    CONSTRAINT uq_leave_type_company_code
        UNIQUE (company_id, code),

    INDEX idx_leave_type_company (company_id),

    CONSTRAINT fk_leave_type_company
        FOREIGN KEY (company_id)
        REFERENCES company(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE leave_request (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    leave_type_id INT NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    back_to_work_date DATE NOT NULL,
    total_day DECIMAL(5,2) NOT NULL DEFAULT 0.00,
    deduct_type_id INT,
    reason TEXT,
    status TINYINT,
    approve_by INT NULL,
    approved_at DATETIME NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
        ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT fk_leave_request_user
        FOREIGN KEY (user_id)
        REFERENCES user(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,

    CONSTRAINT fk_leave_request_leave_type
        FOREIGN KEY (leave_type_id)
        REFERENCES leave_type(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,

    CONSTRAINT fk_leave_request_approver
        FOREIGN KEY (approve_by)
        REFERENCES user(id)
        ON UPDATE CASCADE
        ON DELETE SET NULL,

    INDEX idx_leave_request_user (user_id),
    INDEX idx_leave_request_leave_type (leave_type_id),
    INDEX idx_leave_request_status (status),
    INDEX idx_leave_request_start_date (start_date),
    INDEX idx_leave_request_end_date (end_date),
    INDEX idx_leave_request_approver (approve_by)
) ENGINE=InnoDB
DEFAULT CHARSET=utf8mb4
COLLATE=utf8mb4_unicode_ci;


CREATE TABLE leave_request_detail (

    leave_date DATE NOT NULL,
    deduct_type_id TINYINT UNSIGNED NOT NULL,

    CONSTRAINT fk_lrd_leave_request
        FOREIGN KEY (leave_request_id)
        REFERENCES leave_request(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_lrd_deduct_type
        FOREIGN KEY (deduct_type_id)
        REFERENCES leave_deduct_type(id)
        ON DELETE RESTRICT
);
